    <script>
        function callAPI() {
            const base = Number(document.getElementById("base").value);
            const exponent = Number(document.getElementById("exponent").value);

            if (isNaN(base) || isNaN(exponent)) {
                alert("Please enter valid numbers");
                return;
            }

            fetch("https://hxymtt1zvj.execute-api.us-west-1.amazonaws.com/dev1", {
                method: "POST",
                mode: "cors",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({
                    base: base,
                    exponent: exponent
                })
            })
            .then(response => response.json())
            .then(data => {
                document.getElementById("result").innerText = data.body;
            })
            .catch(error => {
                console.error(error);
                alert("Error calling API");
            });
        }
    </script>
